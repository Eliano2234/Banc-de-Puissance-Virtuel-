import math import time # ========================================== # 1. BIBLIOTHÈQUE VÉHICULES (BASE DE DONNÉES) # ========================================== VEHICLE_DATABASE = { "BMW": { "E46 320d": { "2002": {"puissance_orig": 150, "moteur": "M47TUD20", "masse": 1490, "pneus": "225/45/17", "ratio_rapport": 1.54}, # Exemple de ratio en 3ème }, "E46 330d": { "2003": {"puissance_orig": 204, "moteur": "M57TUD30", "masse": 1595, "pneus": "225/45/17", "ratio_rapport": 1.48}, } }, "SEAT": { "Ibiza Cupra TDI": { "2005": {"puissance_orig": 160, "moteur": "BUK", "masse": 1215, "pneus": "205/40/17", "ratio_rapport": 1.32}, }, "Ibiza TDI FR": { "2004": {"puissance_orig": 130, "moteur": "ASZ", "masse": 1190, "pneus": "205/45/16", "ratio_rapport": 1.32}, } }, "VOLKSWAGEN": { "Golf 4 TDI 150": { "2002": {"puissance_orig": 150, "moteur": "ARL", "masse": 1315, "pneus": "205/55/16", "ratio_rapport": 1.34}, } }, "PEUGEOT": { "307 HDI": { "2005": {"puissance_orig": 90, "moteur": "DV6ATED4", "masse": 1200, "pneus": "195/65/15", "ratio_rapport": 1.37}, }, "1007 HDI": { "2006": {"puissance_orig": 90, "moteur": "DV6ATED4", "masse": 1215, "pneus": "195/50/16", "ratio_rapport": 1.37}, } } } class VirtualDynoApp: def __init__(self): # Variables de configuration modifiables self.masse_vehicule = 0.0 self.ratio_transmission = 0.0 self.taille_pneus = "" self.circonference_roue = 0.0 # État de l'acquisition self.rapport_engage = 3 # Par défaut le rapport 3 est souvent estimé / utilisé self.vitesse_gps = 0.0 self.rpm = 0 self.historique_tirs = [] # ========================================== # 2. SELECTION ET PRE-REMPLISSAGE MODIFIABLE # ========================================== def selectionner_vehicule(self, marque, modele, annee): try: data = VEHICLE_DATABASE[marque][modele][annee] print(f"\n--- Véhicule Sélectionné : {marque} {modele} ({annee}) ---") print(f"Moteur : {data['moteur']} | Puissance d'origine : {data['puissance_orig']} cv") # Pré-remplissage des données self.masse_vehicule = data["masse"] self.ratio_transmission = data["ratio_rapport"] self.taille_pneus = data["pneus"] # Calcul automatique de la circonférence du pneu self.circonference_roue = self._calculer_circonference(self.taille_pneus) self.afficher_parametres_actuels() except KeyError: print("Erreur : Véhicule introuvable dans la bibliothèque.") def _calculer_circonference(self, chaine_pneu): # Exemple : "225/45/17" -> Largeur: 225mm, Flanc: 45%, Jante: 17 pouces parts = chaine_pneu.split('/') largeur = float(parts[0]) flanc_pourcent = float(parts[1]) jante_pouces = float(parts[2]) hauteur_flanc = (largeur * (flanc_pourcent / 100)) / 1000 # en mètres diametre_jante = (jante_pouces * 25.4) / 1000 # en mètres diametre_total = diametre_jante + (2 * hauteur_flanc) return math.pi * diametre_total def modifier_parametre_individuel(self, champ, nouvelle_valeur): """Permet à l'utilisateur de modifier à la main n'importe quelle valeur pré-remplie""" if champ == "masse": self.masse_vehicule = float(nouvelle_valeur) elif champ == "ratio": self.ratio_transmission = float(nouvelle_valeur) elif champ == "pneus": self.taille_pneus = nouvelle_valeur self.circonference_roue = self._calculer_circonference(nouvelle_valeur) print(f"-> Paramètre '{champ}' mis à jour.") def afficher_parametres_actuels(self): print(f"[CONFIGURATION ACTUELLE] Masse: {self.masse_vehicule} kg | Rapport Estimé: {self.ratio_transmission} | Pneus: {self.taille_pneus} (Circ: {self.circonference_roue:.3f}m)") # ========================================== # 3. CONVERSION VITESSE GPS -> RPM VIRTUEL # ========================================== def mettre_a_jour_vitesse_gps(self, vitesse_kmh, rapport_active=False): self.vitesse_gps = vitesse_kmh if rapport_active: # Calcul direct des RPM basé sur la vitesse GPS et la démultiplication # Vitesse (m/min) = (Vitesse km/h * 1000) / 60 vitesse_m_min = (vitesse_kmh * 1000) / 60 rotation_roue_min = vitesse_m_min / self.circonference_roue # RPM Moteur = Rotation Roue * Ratio global du rapport engagé self.rpm = int(rotation_roue_min * self.ratio_transmission * 4.1) # Note : 4.1 simule ici le ratio de pont final else: self.rpm = 0 # ========================================== # 4. PROCÉDURE DE TIR AUTOMATISÉE # ========================================== def lancer_procedure_tir(self): print("\n=== DÉBUT DE LA PROCÉDURE DE TIR ===") print(f"Rapport d'application engagé : Rapport {self.rapport_engage}") # Step 1 : Stabilisation print("\n[ÉTAPPE 1] Stabilisez le régime à 1500 tr/min...") for i in range(3, 0, -1): print(f"Maintien stable... {i}s (RPM virtuels détectés : {self.rpm})") time.sleep(1) # Step 2 : Accélération Pleine Charge print("\n !!! ACCÉLÉREZ À FOND !!! [Pied au plancher]") donnees_tir = [] # Simulation d'une montée en régime (en situation réelle, alimenté par le GPS) sim_vitesse = 45.0 while sim_vitesse < 120.0: self.mettre_a_jour_vitesse_gps(sim_vitesse, rapport_active=True) timestamp = time.time() donnees_tir.append((timestamp, self.rpm, self.vitesse_gps)) print(f" Enregistrement -> {self.rpm} tr/min | {self.vitesse_gps:.1f} km/h") sim_vitesse += 15.0 # Simulation accélération brute time.sleep(0.4) # Step 3 : Fin et Débrayage print("\n[ÉTAPE 3] DÉBRAYEZ ! Lâchez les gaz.") print("=== FIN DE LA PROCÉDURE DE MESURE ===") # Basculement automatique sur le traitement graphique self.generer_et_afficher_graphique(donnees_tir) # ========================================== # 5. GRAPHIQUE ET ANALYSE COMPARATIVE # ========================================== def generer_et_afficher_graphique(self, donnees): print("\n=== ÉCRAN GRAPHIQUE INTERACTIF ===") print("Génération de la courbe [Régime (RPM) / Couple (Nm) / Puissance (ch)]...") # Ici l'algorithme calcule Couple = Force * Rayon de roue, Puissance = Couple * Régime # Stockage pour comparaison future self.historique_tirs.append(donnees) print("-> Graphique enregistré localement dans l'application.") print("-> Graphique exporté et sauvegardé dans la galerie du téléphone (Format Image/PNG).") def option_comparer_origine(self, marque, modele, annee): """Superpose les courbes d'origine constructeur stockées en bibliothèque""" data = VEHICLE_DATABASE[marque][modele][annee] print(f"\n[MODE COMPARATIF ACTIF]") print(f"Superposition active : Courbe constructeur théorique de référence ({data['puissance_orig']} cv) affichée en arrière-plan.") # ========================================== # TEST DU FLUX APPLICATIF UTILISATEUR # ========================================== if __name__ == "__main__": dyno = VirtualDynoApp() # 1. Activation du bouton "Rapport engagé" et lecture GPS en roulant dyno.mettre_a_jour_vitesse_gps(45.0, rapport_active=True) # 4. Lancement de la procédure de tir dyno.lancer_procedure_tir() # 5.
